렌더링 컨텍스트

let _context=Canvas.getContext('2d')

캔버스에서 이루어지는 모든 동작들은 렌더링 컨텍스트 객체의 메소드를 통해 실행됨
당분간은 2차원 그래픽을 사용하므로, 2d 컨텍스트를 사용

기본 도형 그리기

//직사각형
ctx.fillRect(x, y, width, height)//채운 직사각형
ctx.strokeRect(x, y, width, height)//안 채운 직사각형
ctx.clearRect(x, y, width, height)//직사각형 모양으로 비우기
//경로
ctx.beginPath()//경로 초기화
ctx.moveTo(x, y)//그냥 펜만 옮김
ctx.lineTo(x, y)//직선
ctx.quadraticCurveTo(cpx, cpy, x, y)//곡선, 앞의 두 파라미터는 기준점
ctx.stroke();//윤곽선만 그리기
